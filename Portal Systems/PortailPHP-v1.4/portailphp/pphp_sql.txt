CREATE TABLE pphp_docs (
  DO_uid int(8) NOT NULL auto_increment,
  DO_date date NOT NULL default '0000-00-00',
  DO_aut varchar(50) NOT NULL default '',
  DO_rub varchar(50) NOT NULL default '',
  DO_suj varchar(50) NOT NULL default '',
  DO_cont text NOT NULL,
  DO_lect int(8) NOT NULL default '1',

  KEY DO_rub (DO_rub,DO_suj)
) TYPE=MyISAM;
