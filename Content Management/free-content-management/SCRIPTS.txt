USE CMS;

CREATE TABLE cms_t_users 
(UserID VARCHAR(20) NOT NULL, 
Password VARCHAR(20), 
UserName VARCHAR(50), 
EMail VARCHAR(50),
AdminUser CHAR(1),
Active CHAR(1),
PRIMARY KEY (UserID));

CREATE TABLE cms_t_menus (
    ID MEDIUMINT  UNSIGNED NOT NULL AUTO_INCREMENT,
    Title VARCHAR(255),
    HyperLink VARCHAR(255),
    DisplaySequence MEDIUMINT  UNSIGNED,
	Active CHAR(1),
    PRIMARY KEY (id)
);
