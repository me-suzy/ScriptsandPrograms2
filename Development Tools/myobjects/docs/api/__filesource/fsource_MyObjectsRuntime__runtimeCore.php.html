    297. 

  297.     * Makes a selection query.

  298.     *
    299. 

  299.     * <p>Makes a selection query based on the provided table name. It uses the supplied
    300. 

  300.     * uniqueValues in the where clause of the sql query and selects the provided fields.</p>
    301. 

  301.     *
    302. 

  302.     * Example usage :
    303. 

  303.     * <code>
    304. 

  304.     * $result = DbModel::select('users', array('username'=>'eternity'), 'userid, name');
    305. 

  305.     * $users = DbModel::get($result);
    306. 

  306.     * </code>
    307. 

  307.     *
    308. 

  308.     * <p>If you want to add pattern matching in your query you can add "LIKE::"
    309. 

  309.     * and "NOT LIKE::" keywords as prefixes to the values in your arrays.
    310. 

  310.     * For example if you'd like to select users whose name starts with E:</p>
    311. 

  311.     * <code>
    312. 

  312.     * $result = DbModel::select('users', array('name'=>'LIKE::E%'));
    313. 

  313.     * $users = DbModel::get($result);
    314. 

  314.     * </code>

  315.     * For the users whose names do not start with E
    316. 

  316.     * <code>
    317. 

  317.     * $result = DbModel::select('users', array('name'=>'NOT LIKE::E%'));
    318. 

  318.     * $users = DbModel::get($result);
    319. 

  319.     * </code>
    320. 

  320.     * If you want to add != to the query you can use the 'NOT::' prefix
    321. 

  321.     * <code>
    322. 

  322.     * $result = DbModel::select('users', array('username'=>'NOT::erdinc'));
    323. 

  323.     * $users = DbModel::get($result);
    324. 

  324.     * </code>
    325. 

  325.     *
    326. 

  326.     * If you want to add OR statements in your sql, you can write the value part of
    327. 

  327.     * your unique values array also as an array.
    328. 

  328.     * Here is an example:
    329. 

  329.     * <code>
    330. 

  330.     * $result = DbModel::select('users', array('name'=>array('LIKE::E%', 'LIKE::F%')));
    331. 

  331.     * $users = DbModel::get($result);
    332. 

  332.     * </code>
    333. 

  333.     * The above code will select the users whose names begin with E or F.
    334. 

  334.     * <p>You can add more than one constraints for a single field too:</p>
    335. 

  335.     * <code>
    336. 

  336.     * $result = DbModel::select('users', array('name'=>'LIKE::E%', 'name'=>'NOT::Erdinc'));
    337. 

  337.     * $users = DbModel::get($result);
    338. 

  338.     * </code>
    339. 

  339.     * @param string $tableName The table name that will be used for selection query
    340. 

  340.     * @param array $uniqueValues Associated array of unique values that will be used
    341. 

  341.     *  in the where clause.
    342. 

  342.     *  If uniqueValues parameter is not specified no Where clause will be included
    343. 

  343.     *  in the database query.
    344. 

  344.     *  In the value part of your arrays you can add "LIKE::", "NOT LIKE::" or "NOT::"
    345. 

  345.     *  prefixes for pattern matching and inverting.
    346. 

  346.     * @param string $fields Comma seperated values of fields that will be selected.
    347. 

  347.     *  Default value is *
    348. 

  348.     * @param int $limit This parameter should be passed a value larger than 0 to
    349. 

  349.     *  limit the number of rows returned
    350. 

  350.     * @param int $offset The offset that will be used in the limit block of the query
    351. 

  351.     * @return mysqli_result Returns the mysqli result set object of the query
    352. 

  352.     */
    353. 

  353.     public static function select($tableName, $uniqueValues = null, $fields = "*",
    354. 

  354.                                   $limit = 0, $offset = 0) {
    355. 

  355.         // Check if the supplied $uniqueValues variable is an array
    356. 

  356.         if(!is_null($uniqueValues) && !is_array($uniqueValues)) {
    357. 

  357.             throw new IllegalArgumentException('The second parameter passed to 
    358. 

  358.             DbModel::select should be an associative array');
    359. 

  359.         }
    360. 

  360.         // Get the database connection object
    361. 

  361.         $db = MyObjectsBase::getInstance()->getDbConnection();
    362. 

  362.         
    363. 

  363.         // Create the sql query
    364. 

  364.         $sql = 'SELECT ' . $fields . ' FROM `' . $tableName.'`';
    365. 

  365.         if (count($uniqueValues) > 0 ) {
    366. 

  366.             $sql .= ' WHERE ';
    367. 

  367.             $addAnd = false;
    368. 

  368.             foreach ($uniqueValues as $key=>$value) {
    369. 

  369.                 if(!is_array($value)) {
    370. 

  370.                     $value = array($value);
    371. 

  371.                 }
    372. 

  372.                 if($addAnd) $sql .= ' AND';
    373. 

  373.                 $sql .= ' ( ';
    374. 

  374.                 $addOr = false;
    375. 

  375.                 foreach ($value as $val) {
    376. 

  376.                     if(strpos($val, 'LIKE::') === 0) {
    377. 

  377.                         $equalityType = 'LIKE';
    378. 

  378.                         $val = substr($val, 6);
    379. 

  379.                     }
    380. 

  380.                     elseif (strpos($val, 'NOT LIKE::') === 0) {
    381. 

  381.                         $equalityType = 'NOT LIKE';
    382. 

  382.                         $val = substr($val, 10);
    383. 

  383.                     }
    384. 

  384.                     elseif (strpos($val, 'NOT::') === 0) {
    385. 

  385.                         $equalityType = '!=';
    386. 

  386.                         $val = substr($val, 5);
    387. 

  387.                     }
    388. 

  388.                     else {
    389. 

  389.                         $equalityType = '=';
    390. 

  390.                     }
    391. 

  391.                     if($addOr) {
    392. 

  392.                         $sql .= ' OR ';
    393. 

  393.                     }
    394. 

  394.                     $sql .= '`' . $key . '` '.$equalityType.' \'' . $db->real_escape_string($val) . '\' ';
    395. 

  395.                     $addOr = true;
    396. 

  396.                 }
    397. 

  397.                 $sql .= ' ) ';
    398. 

  398.                 $addAnd = true;
    399. 

  399.             }
    400. 

  400.         }
    401. 

  401.         
    402. 

  402.         // If the limit parameter is passed add the limit statement to the query
    403. 

  403.         if($limit > 0) {
    404. 

  404.             $sql .= ' LIMIT ' . $offset . ', ' . $limit; 
    405. 

  405.         }
    406. 

  406.  
    407. 

  407.         // Execute the query and return the result set
    408. 

  408.         return $db->query($sql);
    409. 

  409.     }
