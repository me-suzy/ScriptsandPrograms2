  279.     public function __construct() {

  280.         // Check if a database schema file is registered with the user session

  281.         if(isset($_SESSION['workFolder']) && file_exists(MYOBJECTS_ROOT . '/webclient/tmp/' . $_SESSION['workFolder'] . '/schema.xml')) {
    282.

  283.             $doc = new DOMDocument();

    286. 

  286.                 throw new Exception("Current database schema file is corrupt.");
    287. 

  287.             }
    288. 

  288.             
    289. 

  289.             // Create a DDL object

  290.             $this->ddl = new DDL();
    291. 

  291.             // Gather the info stored in the ddl file into the DDL object
    292. 

  292.             $this->ddl->parse($doc);

  293.         }
    294. 

  294.     }
