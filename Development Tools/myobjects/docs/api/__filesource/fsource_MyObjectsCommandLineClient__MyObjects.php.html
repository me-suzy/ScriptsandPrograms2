  224.     public function proccessProjectFile($projectFile) {
    225. 

  225.         $doc = new DOMDocument();
    226. 

  226.         $doc->load($projectFile);
    227. 

  227.         
    228. 

  228.         // Validate the specified project file using the XML Schema
    229. 

  229.         if(!$doc->schemaValidate(MYOBJECTS_ROOT . '/project.xsd')) {
    230. 

  230.             error($projectFile . ' is not a valid project file');
    231. 

  231.             exit;
    232. 

  232.         }
    233. 

  233.         
    234. 

  234.         $project = $doc->documentElement;
    235. 

  235.         foreach ($project->childNodes as $taskElement) {
    236. 

  236.             if (!($taskElement instanceof DOMElement)) {
    237. 

  237.                 continue;
    238. 

  238.             }
    239. 

  239.             
    240. 

  240.             switch ($taskElement->tagName) {
    241. 

  241.                 case 'generateClasses':
    242. 

  242.                     $this->addGcTask($taskElement);
    243. 

  243.                     break;
    244. 

  244.                 case 'generateSchema':
    245. 

  245.                     $this->addGsTask($taskElement);
    246. 

  246.                     break;

  247.             }

  248.         }

  399.     public function proccess() {
    400. 

  400.         while(!file_exists($this->ddlFile)) {
    401. 

  401.             out("Specified database schema file '".$this->ddlFile."' could not be found. Please specifiy the full path of database schema file, type SKIP to skip the task or type EXIT to exit the application.\n");
    402. 

  402.             $this->ddlFile = read('Input full database schema file path: ');
    403. 

  403.             if(strtolower($this->ddlFile) == 'exit') {
    404. 

  404.                 exit();
    405. 

  405.             }
    406. 

  406.             else if(strtolower($this->ddlFile) == 'skip') {
    407. 

  407.                 break;
    408. 

  408.             }
    409. 

  409.         }
    410. 

  410.         
    411. 

  411.         if(!file_exists($this->outputDir)) {
    412. 

  412.             out('Output directory: ' . $this->outputDir . ' does not exists. Do you want to create it ? (Yes/No)');
    413. 

  413.             $response = read();
    414. 

  414.             $response = strtolower($response);
    415. 

  415.             if($response == 'yes' || $response == 'y') {
    416. 

  416.                 mkdir($this->outputDir, null, true);
    417. 

  417.             } else {
    418. 

  418.                 out('Exiting...');
    419. 

  419.                 exit;
    420. 

  420.             }
    421. 

  421.         }
    422. 

  422.         
    423. 

  423.         $generator = new DefaultClassGenerator($this->getters, $this->setters);
    424. 

  424.         if($this->authorName) {
    425. 

  425.             $generator->setAuthor($this->authorName);
